# fix the KDA (since realisitically the most kills you can get is 30)
def fixDataFrame(df, cname, MAX_KILLS, MAX_ASSISTS):
    df[cname[0]] = [min(MAX_KILLS, x) for x in df[cname[1]]]
    df[cname[1]] = [min(MAX_KILLS + MAX_ASSISTS, x) for x in df[cname[1]]]
    
    # subset only games greater than 5 to avoid outliers
    df = df[df[cname[2]] > 2]

    # eliminate supports and jungler role (will expand on this)
    df = df[df[cname[3]] > 60]
    return df

from sklearn.tree import DecisionTreeClassifier
import sklearn.linear_model
import sklearn.svm

def plot_decision_surface(clf, X_train, Y_train, title):
    plot_step=0.1
    
    if X_train.shape[1] != 2:
        raise ValueError("X_train should have exactly 2 columnns!")
    
    x_min, x_max = X_train[:, 0].min() - plot_step, X_train[:, 0].max() + plot_step
    y_min, y_max = X_train[:, 1].min() - plot_step, X_train[:, 1].max() + plot_step
    xx, yy = np.meshgrid(np.arange(x_min, x_max, plot_step),
                         np.arange(y_min, y_max, plot_step))

    clf.fit(X_train,Y_train)
    if hasattr(clf, 'predict_proba'):
        Z = clf.predict_proba(np.c_[xx.ravel(), yy.ravel()])[:,1]
    else:
        Z = clf.predict(np.c_[xx.ravel(), yy.ravel()])    
    Z = Z.reshape(xx.shape)
    cs = plt.contourf(xx, yy, Z, cmap=plt.cm.Reds)
    plt.scatter(X_train[:,0],X_train[:,1],c=Y_train,cmap=plt.cm.Paired)
    plt.title(title)
    plt.xlabel("win rate")
    plt.ylabel("kd")
    plt.scatter([], [], color = 'grey', marker = 'o', label = 'loss')
    plt.scatter([], [], color = 'crimson', marker = 'o', label = 'win')
    plt.legend(bbox_to_anchor=(1.12, 1), loc=1, borderaxespad=0.)
    plt.show()
    
full_sub = full_df[["winRate", "KD"]]
X_sub = full_sub.as_matrix()
dTree = DecisionTreeClassifier()
dTree.fit(X_sub, Y)
plot_decision_surface(dTree, X_sub, Y, "Decision Tree Decision Surface")

forest_sub = RandomForestClassifier(n_estimators = 15) 
forest_sub.fit(X_sub, Y)
plot_decision_surface(forest_sub, X_sub, Y, "Random Forest Decision Surface")

svm = sklearn.svm.SVC(C = 100, gamma = 1.0, probability = True)
svm.fit(X_sub, Y)
plot_decision_surface(svm, X_sub, Y, "SVM Decision Surface")

These predictions are slightly better than the data with fewer features. Having more features improves the cross validation accuracy slightly. However we didn't expect a much better prediction since a lot of these features are correlated with the other features. For instance, first blood which is the statistic of whether a player gets the first kills during a game is clearly correlated with factors like kill death ratio.

Nevertheless, the random forest decision surface is very similar to the model for the data with fewer features. Surprisingly for both of them, a slightly lower than average historic Kill-Death ratio and a higher historic win rate are the best features for winning a game. In terms of the game, a self-less team oriented player is one who is likely to succeed. The shatters the preconception that one has to singly "carry" through getting kills and money.

4.5 Final Words¶

A prediction accuracy of around 62% on the test set is not too impressive. However, for a game as random League of Legends and for data that is hardly distinguishable, we are still satisfied with how the decision forest performs.

The random forest predictor seems to be the best for categorical data described by multiple features. A KNN or SVM will not work due to the overlap of our data. We do not expect a high precision or recall. Nevertheless, our predictor allows us to make better predictions better than just randomly guessing. Dedicated gamers interested in improving their gameplay as well as casual player looking to "climb the ladder" would be interested in this data.
